78 void RestHandler::get(http_request message) {
79  spdlog::debug("Received GET request {}", message.to_string() );
80  auto paths = uri::split_path(uri::decode(message.relative_uri().path()));
81  if (_require_bearer_token &&
82  (message.headers()["Authorization"] != "Bearer " + _api_key)) {
83  message.reply(status_codes::Unauthorized);
84  return;
85  }
86 
87  if (paths.empty()) {
88  message.reply(status_codes::NotFound);
89  } else {
90  if (paths[0] == "status") {
91  auto state = value::object();
92 
93  switch (_state) {
94  case searching:
95  state["state"] = value::string("searching");
96  break;
97  case syncing:
98  state["state"] = value::string("syncing");
99  break;
100  case processing:
101  state["state"] = value::string("synchronized");
102  break;
103  }
104 
105  state["nof_prb"] = value(_phy.cell().nof_prb);
106  state["cell_id"] = value(_phy.cell().id);
107  state["cfo"] = value(_phy.cfo());
108  state["subcarrier_spacing"] = value(_phy.mbsfn_subcarrier_spacing_khz());
109  message.reply(status_codes::OK, state);
110  } else if (paths[0] == "sdr_params") {
111  value sdr = value::object();
112  sdr["frequency"] = value(_lime.get_frequency());
113  sdr["gain"] = value(_lime.get_gain());
114  sdr["filter_bw"] = value(_lime.get_filter_bw());
115  sdr["antenna"] = value(_lime.get_antenna());
116  sdr["sample_rate"] = value(_lime.get_sample_rate());
117  sdr["buffer_level"] = value(_lime.get_buffer_level());
118  message.reply(status_codes::OK, sdr);
119  } else if (paths[0] == "ce_values") {
120  auto cestream = Concurrency::streams::bytestream::open_istream(_ce_values);
121  message.reply(status_codes::OK, cestream);
122  } else if (paths[0] == "pdsch_status") {
123  value sdr = value::object();
124  sdr["bler"] = value(static_cast<float>(_pdsch.errors) /
125  static_cast<float>(_pdsch.total));
126  sdr["ber"] = value(_pdsch.ber);
127  sdr["mcs"] = value(_pdsch.mcs);
128  sdr["present"] = 1;
129  message.reply(status_codes::OK, sdr);
130  } else if (paths[0] == "pdsch_data") {
131  auto cestream = Concurrency::streams::bytestream::open_istream(_pdsch.data);
132  message.reply(status_codes::OK, cestream);
133  } else if (paths[0] == "mcch_status") {
134  value sdr = value::object();
135  sdr["bler"] = value(static_cast<float>(_mcch.errors) /
136  static_cast<float>(_mcch.total));
137  sdr["ber"] = value("-");
138  sdr["mcs"] = value(_mcch.mcs);
139  sdr["present"] = 1;
140  message.reply(status_codes::OK, sdr);
141  } else if (paths[0] == "mcch_data") {
142  auto cestream = Concurrency::streams::bytestream::open_istream(_mcch.data);
143  message.reply(status_codes::OK, cestream);
144  } else if (paths[0] == "mch_info") {
145  std::vector<value> mi;
146  auto mch_info = _phy.mch_info();
147  std::for_each(std::begin(mch_info), std::end(mch_info), [&mi](Phy::mch_info_t const& mch) {
148  value m;
149  m["mcs"] = value(mch.mcs);
150  std::vector<value> mti;
151  std::for_each(std::begin(mch.mtchs), std::end(mch.mtchs), [&mti](Phy::mtch_info_t const& mtch) {
152  value mt;
153  mt["tmgi"] = value(mtch.tmgi);
154  mt["dest"] = value(mtch.dest);
155  mt["lcid"] = value(mtch.lcid);
156  mti.push_back(mt);
157  });
158  m["mtchs"] = value::array(mti);
159  mi.push_back(m);
160  });
161  message.reply(status_codes::OK, value::array(mi));
162  } else if (paths[0] == "mch_status") {
163  int idx = paths[1] == "0" ? 0 : 1;
164  value sdr = value::object();
165  sdr["bler"] = value(static_cast<float>(_mch[idx].errors) /
166  static_cast<float>(_mch[idx].total));
167  sdr["ber"] = value("-");
168  sdr["mcs"] = value(_mch[idx].mcs);
169  sdr["present"] = value(_mch[idx].present);
170  message.reply(status_codes::OK, sdr);
171  } else if (paths[0] == "mch_data") {
172  int idx = paths[1] == "0" ? 0 : 1;
173  auto cestream = Concurrency::streams::bytestream::open_istream(_mch[idx].data);
174  message.reply(status_codes::OK, cestream);
175  } else if (paths[0] == "log") {
176  std::string logfile = "/tmp/ofr.log";
177  _cfg.lookupValue("app.log_file", logfile);
178 
179  Concurrency::streams::file_stream<uint8_t>::open_istream(logfile).then(
180  [message](const Concurrency::streams::basic_istream<unsigned char>&
181  file_stream) {
182  message.reply(status_codes::OK, file_stream, "text/plain");
183  });
184  }
185  }

188 void RestHandler::put(http_request message) {
189  spdlog::debug("Received PUT request {}", message.to_string() );
190 
191  if (_require_bearer_token &&
192  (message.headers()["Authorization"] != "Bearer " + _api_key)) {
193  message.reply(status_codes::Unauthorized);
194  return;
195  }
196 
197  auto paths = uri::split_path(uri::decode(message.relative_uri().path()));
198  if (paths.empty()) {
199  message.reply(status_codes::NotFound);
200  } else {
201  if (paths[0] == "sdr_params") {
202  value answer;
203 
204  const auto & jval = message.extract_json().get();
205 
206  _set_params(
207  jval.at("antenna").serialize(),
208  jval.at("frequency").as_integer(),
209  jval.at("gain").as_double(),
210  jval.at("sample_rate").as_integer(),
211  jval.at("filter_bw").as_integer());
212 
213  message.reply(status_codes::OK, answer);
214  }
215  }
