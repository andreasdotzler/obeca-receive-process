77 void RestHandler::get(http_request message) {
78  spdlog::debug("Received GET request {}", message.to_string() );
79  auto paths = uri::split_path(uri::decode(message.relative_uri().path()));
80  if (_require_bearer_token &&
81  (message.headers()["Authorization"] != "Bearer " + _api_key)) {
82  message.reply(status_codes::Unauthorized);
83  return;
84  }
85 
86  if (paths.empty()) {
87  message.reply(status_codes::NotFound);
88  } else {
89  if (paths[0] == "status") {
90  auto state = value::object();
91 
92  switch (_state) {
93  case searching:
94  state["state"] = value::string("searching");
95  break;
96  case syncing:
97  state["state"] = value::string("syncing");
98  break;
99  case processing:
100  state["state"] = value::string("synchronized");
101  break;
102  }
103 
104  if (_phy.cell().nof_prb == _phy.cell().mbsfn_prb) {
105  state["nof_prb"] = value(_phy.cell().nof_prb);
106  } else {
107  state["nof_prb"] = value(_phy.cell().mbsfn_prb);
108  }
109  state["cell_id"] = value(_phy.cell().id);
110  state["cfo"] = value(_phy.cfo());
111  state["cinr_db"] = value(cinr_db());
112  state["subcarrier_spacing"] = value(_phy.mbsfn_subcarrier_spacing_khz());
113  message.reply(status_codes::OK, state);
114  } else if (paths[0] == "sdr_params") {
115  value sdr = value::object();
116  sdr["frequency"] = value(_lime.get_frequency());
117  sdr["gain"] = value(_lime.get_gain());
118  sdr["filter_bw"] = value(_lime.get_filter_bw());
119  sdr["antenna"] = value(_lime.get_antenna());
120  sdr["sample_rate"] = value(_lime.get_sample_rate());
121  sdr["buffer_level"] = value(_lime.get_buffer_level());
122  message.reply(status_codes::OK, sdr);
123  } else if (paths[0] == "ce_values") {
124  auto cestream = Concurrency::streams::bytestream::open_istream(_ce_values);
125  message.reply(status_codes::OK, cestream);
126  } else if (paths[0] == "pdsch_status") {
127  value sdr = value::object();
128  sdr["bler"] = value(static_cast<float>(_pdsch.errors) /
129  static_cast<float>(_pdsch.total));
130  sdr["ber"] = value(_pdsch.ber);
131  sdr["mcs"] = value(_pdsch.mcs);
132  sdr["present"] = 1;
133  message.reply(status_codes::OK, sdr);
134  } else if (paths[0] == "pdsch_data") {
135  auto cestream = Concurrency::streams::bytestream::open_istream(_pdsch.GetData());
136  message.reply(status_codes::OK, cestream);
137  } else if (paths[0] == "mcch_status") {
138  value sdr = value::object();
139  sdr["bler"] = value(static_cast<float>(_mcch.errors) /
140  static_cast<float>(_mcch.total));
141  sdr["ber"] = value(_mcch.ber);
142  sdr["mcs"] = value(_mcch.mcs);
143  sdr["present"] = 1;
144  message.reply(status_codes::OK, sdr);
145  } else if (paths[0] == "mcch_data") {
146  auto cestream = Concurrency::streams::bytestream::open_istream(_mcch.GetData());
147  message.reply(status_codes::OK, cestream);
148  } else if (paths[0] == "mch_info") {
149  std::vector<value> mi;
150  auto mch_info = _phy.mch_info();
151  std::for_each(std::begin(mch_info), std::end(mch_info), [&mi](Phy::mch_info_t const& mch) {
152  value m;
153  m["mcs"] = value(mch.mcs);
154  std::vector<value> mti;
155  std::for_each(std::begin(mch.mtchs), std::end(mch.mtchs), [&mti](Phy::mtch_info_t const& mtch) {
156  value mt;
157  mt["tmgi"] = value(mtch.tmgi);
158  mt["dest"] = value(mtch.dest);
159  mt["lcid"] = value(mtch.lcid);
160  mti.push_back(mt);
161  });
162  m["mtchs"] = value::array(mti);
163  mi.push_back(m);
164  });
165  message.reply(status_codes::OK, value::array(mi));
166  } else if (paths[0] == "mch_status") {
167  int idx = std::stoi(paths[1]);
168  value sdr = value::object();
169  sdr["bler"] = value(static_cast<float>(_mch[idx].errors) /
170  static_cast<float>(_mch[idx].total));
171  sdr["ber"] = value(_mch[idx].ber);
172  sdr["mcs"] = value(_mch[idx].mcs);
173  sdr["present"] = value(_mch[idx].present);
174  message.reply(status_codes::OK, sdr);
175  } else if (paths[0] == "mch_data") {
176  int idx = std::stoi(paths[1]);
177  auto cestream = Concurrency::streams::bytestream::open_istream(_mch[idx].GetData());
178  message.reply(status_codes::OK, cestream);
179  } else if (paths[0] == "log") {
180  std::string logfile = "/var/log/syslog";
181 
182  Concurrency::streams::file_stream<uint8_t>::open_istream(logfile).then(
183  [message](const Concurrency::streams::basic_istream<unsigned char>&
184  file_stream) {
185  message.reply(status_codes::OK, file_stream, "text/plain");
186  });
187  }
188  }
189 }

191 void RestHandler::put(http_request message) {
192  spdlog::debug("Received PUT request {}", message.to_string() );
193 
194  if (_require_bearer_token &&
195  (message.headers()["Authorization"] != "Bearer " + _api_key)) {
196  message.reply(status_codes::Unauthorized);
197  return;
198  }
199 
200  auto paths = uri::split_path(uri::decode(message.relative_uri().path()));
201  if (paths.empty()) {
202  message.reply(status_codes::NotFound);
203  } else {
204  if (paths[0] == "sdr_params") {
205  value answer;
206 
207  const auto & jval = message.extract_json().get();
208 
209  _set_params(
210  jval.at("antenna").as_string(),
211  jval.at("frequency").as_integer(),
212  jval.at("gain").as_double(),
213  jval.at("sample_rate").as_integer(),
214  jval.at("filter_bw").as_integer());
215 
216  message.reply(status_codes::OK, answer);
217  }
218  }
