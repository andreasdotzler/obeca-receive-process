55 auto Phy::synchronize_subframe() -> bool {
56  int ret = srslte_ue_sync_zerocopy(&_ue_sync, _mib_buffer, _buffer_max_samples); // NOLINT
57  if (ret < 0) {
58  spdlog::error("SYNC: Error calling ue_sync_get_buffer.\n");
59  return false;
60  }
61 
62  if (ret == 1) {
63  std::array<uint8_t, SRSLTE_BCH_PAYLOAD_LEN> bch_payload = {};
64  if (srslte_ue_sync_get_sfidx(&_ue_sync) == 0) {
65  int sfn_offset = 0;
66  int n =
67  srslte_ue_mib_decode(&_mib, bch_payload.data(), nullptr, &sfn_offset);
68  if (n == 1) {
69  uint32_t sfn = 0;
70  srslte_pbch_mib_mbms_unpack(bch_payload.data(), &_cell, &sfn, nullptr,
71  _override_nof_prb);
72  sfn = (sfn + sfn_offset * kSfnOffset) % kMaxSfn;
73  _tti = sfn * kSubframesPerFrame;
74  return true;
75  }
76  }
77  }
78  return false;
79 }

81 auto Phy::cell_search() -> bool {
82  std::array<srslte_ue_cellsearch_result_t, kMaxCellsToDiscover> found_cells = {0};
83 
84  uint32_t max_peak_cell = 0;
85  int ret = srslte_ue_cellsearch_scan(&_cell_search, found_cells.data(), &max_peak_cell);
86  if (ret < 0) {
87  spdlog::error("Phy: Error decoding MIB: Error searching PSS\n");
88  return false;
89  }
90  if (ret == 0) {
91  spdlog::error("Phy: Could not find any cell in this frequency\n");
92  return false;
93  }
94 
95  srslte_cell_t new_cell = {};
96  new_cell.id = found_cells.at(max_peak_cell).cell_id;
97  new_cell.cp = found_cells.at(max_peak_cell).cp;
98  new_cell.frame_type = found_cells.at(max_peak_cell).frame_type;
99  new_cell.mbms_dedicated = true;
100  float cfo = found_cells.at(max_peak_cell).cfo;
101 
102  spdlog::info("Phy: PSS/SSS detected: Mode {}, PCI {}, CFO {} KHz, CP {}",
103  new_cell.frame_type != 0U ? "TDD" : "FDD", new_cell.id,
104  cfo / 1000, srslte_cp_string(new_cell.cp));
105 
106  if (srslte_ue_mib_sync_set_cell_prb(&_mib_sync, new_cell, _cs_nof_prb) != 0) {
107  spdlog::error("Phy: Error setting UE MIB sync cell");
108  return false;
109  }
110 
111  srslte_ue_sync_reset(&_mib_sync.ue_sync);
112 
113  std::array<uint8_t, SRSLTE_BCH_PAYLOAD_LEN> bch_payload = {};
114  /* Find and decode MIB */
115  int sfn_offset = 0;
116  ret = srslte_ue_mib_sync_decode_prb(&_mib_sync, 40, bch_payload.data(), &new_cell.nof_ports, &sfn_offset, _cs_nof_prb);
117  if (ret == 1) {
118  uint32_t sfn = 0;
119  srslte_pbch_mib_mbms_unpack(bch_payload.data(), &new_cell, &sfn, nullptr,
120  _override_nof_prb);
121 
122  spdlog::info(
123  "Phy: MIB Decoded. Mode {}, PCI {}, PRB {}, Ports {}, CFO {} KHz, SFN "
124  "{}\n",
125  new_cell.frame_type != 0u ? "TDD" : "FDD", new_cell.id,
126  new_cell.nof_prb, new_cell.nof_ports, cfo / 1000, sfn);
127 
128  if (!srslte_cell_isvalid(&new_cell)) {
129  spdlog::error("SYNC: Detected invalid cell.\n");
130  return false;
131  }
132 
133  _cell = new_cell;
134  if (srslte_ue_sync_set_cell(&_ue_sync, _cell) != 0) {
135  spdlog::error("Phy: failed to set cell.\n");
136  return false;
137  }
138  if (srslte_ue_mib_set_cell(&_mib, _cell) != 0) {
139  spdlog::error("Phy: Error setting UE MIB cell");
140  return false;
141  }
142 
143  return true;
144  }
145 
146  spdlog::error("Phy: failed to receive MIB\n");
147  return false;
148 }

150 auto Phy::init() -> bool {
151  if (srslte_ue_cellsearch_init_multi_prb_cp(&_cell_search, 8, receive_callback, 1,
152  this, _cs_nof_prb, true) != 0) {
153  spdlog::error("Phy: error while initiating UE cell search\n");
154  return false;
155  }
156  srslte_ue_cellsearch_set_nof_valid_frames(&_cell_search, 4);
157 
158  if (srslte_ue_sync_init_multi(&_ue_sync, MAX_PRB, false, receive_callback, 1,
159  this) != 0) {
160  spdlog::error("Cannot init ue_sync");
161  return false;
162  }
163 
164  if (srslte_ue_mib_sync_init_multi_prb(&_mib_sync, receive_callback, 1, this,
165  _cs_nof_prb) != 0) {
166  spdlog::error("Cannot init ue_mib_sync");
167  return false;
168  }
169 
170  if (srslte_ue_mib_init(&_mib, _mib_buffer[0], 50) != 0) {
171  spdlog::error("Cannot init ue_mib");
172  return false;
173  }
174 
175  return true;
176 }

207 void Phy::set_mbsfn_config(const srslte::mcch_msg_t& mcch) {
208  _mcch = mcch;
209  _mch_configured = true;
210 
211  _mch_info.clear();

213  mch_info_t mch_info;
214  mch_info.mcs = _mcch.pmch_info_list[i].data_mcs;

222  }
223  tmgi += _mcch.pmch_info_list[i].mbms_session_info_list[j].tmgi.plmn_id.explicit_value.to_string();
224  mtch_info.tmgi = tmgi;
225  mtch_info.dest = _dests[mch_info.mcs][mtch_info.lcid];

233 auto Phy::mbsfn_config_for_tti(uint32_t tti, unsigned& area)
234  -> srslte_mbsfn_cfg_t {
235  srslte_mbsfn_cfg_t cfg;
236  cfg.enable = false;
237  cfg.is_mcch = false;
238 
239  if (!_mcch_configured) {
240  {
241  return cfg;
242  }
243  }
244 
245  uint32_t sfn = tti / 10;
246  uint8_t sf = tti % 10;
247 
248  srslte::mbsfn_area_info_t& area_info = _sib13.mbsfn_area_info_list[0];
249 
250  cfg.mbsfn_area_id = area_info.mbsfn_area_id;
251  cfg.non_mbsfn_region_length = enum_to_number(area_info.non_mbsfn_region_len);
252 
253  if (sfn % enum_to_number(area_info.mcch_cfg.mcch_repeat_period) == area_info.mcch_cfg.mcch_offset &&
254  _mcch_table[sf] == 1) {
255  // MCCH
256  if (_decode_mcch) {
257  cfg.mbsfn_mcs = enum_to_number(area_info.mcch_cfg.sig_mcs);
258  cfg.enable = true;
259  cfg.is_mcch = true;
260  }
261  } else {
262  if (_mch_configured) {
263  cfg.mbsfn_area_id = area_info.mbsfn_area_id;
264 
265  for (uint32_t i = 0; i < _mcch.nof_pmch_info; i++) {
266  unsigned fn_in_scheduling_period = sfn % enum_to_number(_mcch.pmch_info_list[i].mch_sched_period);
267  unsigned sf_idx = fn_in_scheduling_period * 10 + sf - (fn_in_scheduling_period / 4) - 1;
268 
269  spdlog::debug("tti {}, fn_in_ {}, sf_idx {}", tti, fn_in_scheduling_period, sf_idx);
270 
271  if (sf_idx <= _mcch.pmch_info_list[i].sf_alloc_end) {
272  area = i;
273  if ((i == 0 && fn_in_scheduling_period == 0 && sf == 1) ||
274  (i > 0 && _mcch.pmch_info_list[i-1].sf_alloc_end + 1 == sf_idx)) {
275  spdlog::debug("assigning sig_mcs {}", area_info.mcch_cfg.sig_mcs);
276  cfg.mbsfn_mcs = enum_to_number(area_info.mcch_cfg.sig_mcs);
277  } else {
278  spdlog::debug("assigning pmch_mcs {}", area_info.mcch_cfg.sig_mcs);
279  cfg.mbsfn_mcs = _mcch.pmch_info_list[i].data_mcs;
280  }
281  cfg.enable = true;
282  break;
283  }
284  }
285  }
286  }
287  return cfg;
288 }
