101 static auto parse_opt(int key, char *arg, struct argp_state *state) -> error_t {
102  auto arguments = static_cast<struct arguments *>(state->input);
103  switch (key) {
104  case 'c':

106  break;
107  case 'l':
108  arguments->log_level = static_cast<unsigned>(strtoul(arg, nullptr, 10));

112  static_cast<unsigned>(strtoul(arg, nullptr, 10));
113  break;
114  case 'f':
115  arguments->sample_file = arg;
116  break;
117  case 'w':
118  arguments->write_sample_file = arg;

122  break;
123  case 'p':
124  arguments->override_nof_prb =
125  static_cast<int8_t>(strtol(arg, nullptr, 10));
126  break;
127  case ARGP_KEY_ARG:
128  argp_usage(state);
129  break;
130  default:
131  return ARGP_ERR_UNKNOWN;
132  }
